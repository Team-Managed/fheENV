# fheENV Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build fheENV — a zero-trust .env secrets manager where secrets are AES-256 encrypted client-side, blobs stored on IPFS, and AES keys stored as FHE ciphertexts on-chain so not even the platform operator can read team secrets.

**Architecture:** A single `fheENVRegistry.sol` contract namespaces all projects. Each environment's AES-256 key is split into two `euint128` FHE handles stored on-chain; the actual encrypted blob lives on IPFS. A Next.js web UI and a Node.js CLI both use `@cofhe/sdk` to encrypt/decrypt client-side. No plaintext secret ever touches any server.

**Tech Stack:** Solidity 0.8.25 + `@fhenixprotocol/cofhe-contracts`, Hardhat + `@cofhe/hardhat-plugin`, Next.js 14 (App Router), wagmi v2, viem, `@cofhe/sdk/web`, `@cofhe/sdk/node`, Node.js CLI (commander.js), AES-256-GCM (Web Crypto / Node `crypto`), IPFS via Pinata, Sepolia testnet.

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.25;

import "@fhenixprotocol/cofhe-contracts/FHE.sol";

/// @title fheENVRegistry
/// @notice Zero-trust secrets registry. Stores AES-256 keys as FHE ciphertexts.
///         The platform operator is cryptographically incapable of reading secrets.
contract fheENVRegistry {

    // ─── Data Structures ──────────────────────────────────────────────────────

    struct Environment {
        euint128 aesKeyHigh;   // upper 16 bytes of AES-256 key (FHE encrypted)
        euint128 aesKeyLow;    // lower 16 bytes of AES-256 key (FHE encrypted)
        string   blobCid;      // IPFS CID of AES-GCM encrypted .env blob
        uint256  version;      // increments on rotation — optimistic locking
        uint256  updatedAt;
        bool     initialized;
    }

    struct Project {
        string   name;
        address  primaryOwner;
        uint256  createdAt;
        bool     exists;
    }

    // ─── State ────────────────────────────────────────────────────────────────

    uint256 public nextProjectId;

    // projectId → Project metadata
    mapping(uint256 => Project) public projects;

    // projectId → envHash → Environment
    // envHash = keccak256(bytes(lowercaseEnvName))
    mapping(uint256 => mapping(bytes32 => Environment)) public environments;

    // projectId → address → isOwner
    mapping(uint256 => mapping(address => bool)) public owners;

    // projectId → envHash → address → hasReadAccess
    mapping(uint256 => mapping(bytes32 => mapping(address => bool))) public members;

    // ─── Events ───────────────────────────────────────────────────────────────

    event ProjectCreated(uint256 indexed projectId, address indexed owner, string name);
    event EnvironmentUpdated(uint256 indexed projectId, bytes32 indexed envHash, string blobCid, uint256 version);
    event AccessGranted(uint256 indexed projectId, bytes32 indexed envHash, address indexed member);
    event AccessRevoked(uint256 indexed projectId, bytes32 indexed envHash, address indexed member);
    event OwnerAdded(uint256 indexed projectId, address indexed newOwner);

    // ─── Modifiers ────────────────────────────────────────────────────────────

    modifier onlyProjectOwner(uint256 projectId) {
        require(owners[projectId][msg.sender], "Not a project owner");
        _;
    }

    modifier projectExists(uint256 projectId) {
        require(projects[projectId].exists, "Project does not exist");
        _;
    }

    // ─── Project Management ───────────────────────────────────────────────────

    /// @notice Create a new project. Caller becomes the primary owner.
    function createProject(string calldata name) external returns (uint256 projectId) {
        require(bytes(name).length > 0 && bytes(name).length <= 64, "Name: 1-64 chars");
        projectId = nextProjectId++;
        projects[projectId] = Project({
            name: name,
            primaryOwner: msg.sender,
            createdAt: block.timestamp,
            exists: true
        });
        owners[projectId][msg.sender] = true;
        emit ProjectCreated(projectId, msg.sender, name);
    }

    /// @notice Add a co-owner who can manage team access and rotate secrets.
    function addOwner(uint256 projectId, address newOwner)
        external
        projectExists(projectId)
        onlyProjectOwner(projectId)
    {
        require(newOwner != address(0), "Invalid address");
        owners[projectId][newOwner] = true;
        emit OwnerAdded(projectId, newOwner);
    }

    // ─── Environment Management ───────────────────────────────────────────────

    /// @notice Normalize env name to hash to prevent case collisions.
    function envNameToHash(string calldata envName) public pure returns (bytes32) {
        return keccak256(bytes(envName));
    }

    /// @notice Push a new encrypted environment. Uses optimistic locking via
    ///         expectedVersion to prevent silent overwrites from concurrent rotations.
    /// @param projectId       Project to update
    /// @param envName         Environment name (e.g. "production") — case-insensitive
    /// @param inKeyHigh       FHE-encrypted upper 16 bytes of AES-256 key
    /// @param inKeyLow        FHE-encrypted lower 16 bytes of AES-256 key
    /// @param blobCid         IPFS CID of the AES-GCM encrypted .env blob
    /// @param expectedVersion Must match current version (0 for first push)
    function updateEnvironment(
        uint256 projectId,
        string calldata envName,
        InEuint128 calldata inKeyHigh,
        InEuint128 calldata inKeyLow,
        string calldata blobCid,
        uint256 expectedVersion
    ) external projectExists(projectId) onlyProjectOwner(projectId) {
        bytes32 envHash = envNameToHash(envName);
        Environment storage env = environments[projectId][envHash];

        require(env.version == expectedVersion, "Version mismatch: concurrent rotation detected");
        require(bytes(blobCid).length > 0, "blobCid required");

        euint128 keyHigh = FHE.asEuint128(inKeyHigh);
        euint128 keyLow  = FHE.asEuint128(inKeyLow);

        // Contract retains access to re-grant to future members
        FHE.allowThis(keyHigh);
        FHE.allowThis(keyLow);

        // Owner retains access
        FHE.allow(keyHigh, msg.sender);
        FHE.allow(keyLow, msg.sender);

        env.aesKeyHigh  = keyHigh;
        env.aesKeyLow   = keyLow;
        env.blobCid     = blobCid;
        env.version     = expectedVersion + 1;
        env.updatedAt   = block.timestamp;
        env.initialized = true;

        emit EnvironmentUpdated(projectId, envHash, blobCid, env.version);
    }

    // ─── Access Control ───────────────────────────────────────────────────────

    /// @notice Grant a single member decrypt access to an environment.
    function grantAccess(
        uint256 projectId,
        string calldata envName,
        address member
    ) external projectExists(projectId) onlyProjectOwner(projectId) {
        require(member != address(0), "Invalid address");
        bytes32 envHash = envNameToHash(envName);
        Environment storage env = environments[projectId][envHash];
        require(env.initialized, "Environment not initialized");

        FHE.allow(env.aesKeyHigh, member);
        FHE.allow(env.aesKeyLow, member);
        members[projectId][envHash][member] = true;

        emit AccessGranted(projectId, envHash, member);
    }

    /// @notice Grant decrypt access to multiple members in a single transaction.
    function batchGrantAccess(
        uint256 projectId,
        string calldata envName,
        address[] calldata newMembers
    ) external projectExists(projectId) onlyProjectOwner(projectId) {
        require(newMembers.length <= 100, "Max 100 members per batch");
        bytes32 envHash = envNameToHash(envName);
        Environment storage env = environments[projectId][envHash];
        require(env.initialized, "Environment not initialized");

        for (uint256 i = 0; i < newMembers.length; i++) {
            address m = newMembers[i];
            require(m != address(0), "Invalid address in batch");
            FHE.allow(env.aesKeyHigh, m);
            FHE.allow(env.aesKeyLow, m);
            members[projectId][envHash][m] = true;
            emit AccessGranted(projectId, envHash, m);
        }
    }

    /// @notice Remove a member from the access list and emit AccessRevoked.
    ///         IMPORTANT: CoFHE has no revokeAllow primitive — this does NOT
    ///         cryptographically invalidate the old FHE handles. The owner MUST
    ///         call updateEnvironment (rotation) immediately after to issue new
    ///         ciphertext handles that the removed member never receives FHE.allow on.
    ///         The CLI enforces this automatically: `fheenv team remove` triggers rotate.
    function revokeAccess(
        uint256 projectId,
        string calldata envName,
        address member
    ) external projectExists(projectId) onlyProjectOwner(projectId) {
        bytes32 envHash = envNameToHash(envName);
        members[projectId][envHash][member] = false;
        emit AccessRevoked(projectId, envHash, member);
        // NOTE: caller must follow up with updateEnvironment() to rotate the AES key.
        // Until rotation, the removed member can still decrypt the OLD ciphertext
        // because FHE.allow is append-only in the current CoFHE ACL contract.
    }

    // ─── Read Functions ───────────────────────────────────────────────────────

    /// @notice Returns environment handles. Caller can only decrypt if permitted.
    function getEnvironment(uint256 projectId, string calldata envName)
        external
        view
        returns (
            euint128 aesKeyHigh,
            euint128 aesKeyLow,
            string memory blobCid,
            uint256 version
        )
    {
        bytes32 envHash = envNameToHash(envName);
        Environment storage env = environments[projectId][envHash];
        require(env.initialized, "Environment not initialized");
        return (env.aesKeyHigh, env.aesKeyLow, env.blobCid, env.version);
    }

    /// @notice Check if an address has access to an environment.
    function hasAccess(uint256 projectId, string calldata envName, address member)
        external view returns (bool)
    {
        return members[projectId][envNameToHash(envName)][member];
    }
}
```

- [ ] **Step 3.1: Write `test/fheENVRegistry.test.ts`**

```typescript
import { expect } from "chai";
import { ethers } from "hardhat";
import hre from "hardhat";
import { CofheClient, Encryptable, FheTypes } from "@cofhe/sdk";

describe("fheENVRegistry", function () {
  let registry: any;
  let owner: any;
  let member: any;
  let stranger: any;
  let cofheClient: CofheClient;

  // AES-256 key split: simulate 32 bytes as two uint128 values
  const AES_KEY_HIGH = BigInt("0xdeadbeefcafebabe1234567890abcdef");
  const AES_KEY_LOW = BigInt("0xfeedfacebadf00d123456789abcdef01");
  const BLOB_CID = "QmTestBlobCID123456789";

  // Encryptable.uint128() is the correct factory — NOT { type, value } objects.
  // Both uint128 values together = 256 bits, well under the 2048-bit per-call limit.
  async function encryptAesKey(client: CofheClient) {
    const [high, low] = await client
      .encryptInputs([
        Encryptable.uint128(AES_KEY_HIGH),
        Encryptable.uint128(AES_KEY_LOW),
      ])
      .execute();
    return { high, low };
  }

  beforeEach(async function () {
    [owner, member, stranger] = await ethers.getSigners();
    // hre.cofhe.createClientWithBatteries creates + connects a CofheClient
    // and pre-generates a self-permit so decryptForView works immediately.
    cofheClient = await hre.cofhe.createClientWithBatteries(owner);

    const Factory = await ethers.getContractFactory("fheENVRegistry");
    registry = await Factory.deploy();
    await registry.waitForDeployment();
  });

  // ── Project Management ────────────────────────────────────────────────────

  it("creates a project and emits event", async function () {
    const tx = await registry.connect(owner).createProject("my-app");
    const receipt = await tx.wait();
    const event = receipt.logs.find(
      (l: any) => l.fragment?.name === "ProjectCreated",
    );
    expect(event.args.owner).to.equal(owner.address);
    expect(event.args.name).to.equal("my-app");
    expect(event.args.projectId).to.equal(0n);
  });

  it("increments projectId for each project", async function () {
    await registry.connect(owner).createProject("project-a");
    await registry.connect(owner).createProject("project-b");
    expect(await registry.nextProjectId()).to.equal(2n);
  });

  it("rejects empty project name", async function () {
    await expect(registry.connect(owner).createProject("")).to.be.revertedWith(
      "Name: 1-64 chars",
    );
  });

  it("allows adding a co-owner", async function () {
    await registry.connect(owner).createProject("my-app");
    await registry.connect(owner).addOwner(0, member.address);
    expect(await registry.owners(0, member.address)).to.equal(true);
  });

  // ── Environment Management ────────────────────────────────────────────────

  it("pushes an environment with version 0 → 1", async function () {
    await registry.connect(owner).createProject("my-app");
    const { high, low } = await encryptAesKey(cofheClient);

    const tx = await registry
      .connect(owner)
      .updateEnvironment(0, "production", high, low, BLOB_CID, 0);
    const receipt = await tx.wait();

    const event = receipt.logs.find(
      (l: any) => l.fragment?.name === "EnvironmentUpdated",
    );
    expect(event.args.blobCid).to.equal(BLOB_CID);
    expect(event.args.version).to.equal(1n);
  });

  it("rejects rotation with wrong expectedVersion (concurrent rotation guard)", async function () {
    await registry.connect(owner).createProject("my-app");
    const { high, low } = await encryptAesKey(cofheClient);

    // First push succeeds (expectedVersion = 0 → version becomes 1)
    await registry
      .connect(owner)
      .updateEnvironment(0, "production", high, low, BLOB_CID, 0);

    // Second push with stale expectedVersion = 0 must revert
    const { high: h2, low: l2 } = await encryptAesKey(cofheClient);
    await expect(
      registry
        .connect(owner)
        .updateEnvironment(0, "production", h2, l2, "QmOther", 0),
    ).to.be.revertedWith("Version mismatch: concurrent rotation detected");
  });

  it("treats env names case-insensitively via hash", async function () {
    const hash1 = await registry.envNameToHash("production");
    const hash2 = await registry.envNameToHash("production");
    expect(hash1).to.equal(hash2);
  });

  it("rejects non-owner pushing environment", async function () {
    await registry.connect(owner).createProject("my-app");
    const { high, low } = await encryptAesKey(cofheClient);
    await expect(
      registry
        .connect(stranger)
        .updateEnvironment(0, "production", high, low, BLOB_CID, 0),
    ).to.be.revertedWith("Not a project owner");
  });

  // ── Access Control ────────────────────────────────────────────────────────

  it("grants access to a member", async function () {
    await registry.connect(owner).createProject("my-app");
    const { high, low } = await encryptAesKey(cofheClient);
    await registry
      .connect(owner)
      .updateEnvironment(0, "production", high, low, BLOB_CID, 0);

    await expect(
      registry.connect(owner).grantAccess(0, "production", member.address),
    ).to.emit(registry, "AccessGranted");

    expect(await registry.hasAccess(0, "production", member.address)).to.equal(
      true,
    );
  });

  it("batch grants access to multiple members", async function () {
    await registry.connect(owner).createProject("my-app");
    const { high, low } = await encryptAesKey(cofheClient);
    await registry
      .connect(owner)
      .updateEnvironment(0, "production", high, low, BLOB_CID, 0);

    const [, m1, m2, m3] = await ethers.getSigners();
    await registry
      .connect(owner)
      .batchGrantAccess(0, "production", [m1.address, m2.address, m3.address]);

    expect(await registry.hasAccess(0, "production", m1.address)).to.equal(
      true,
    );
    expect(await registry.hasAccess(0, "production", m2.address)).to.equal(
      true,
    );
    expect(await registry.hasAccess(0, "production", m3.address)).to.equal(
      true,
    );
  });

  it("revokes access", async function () {
    await registry.connect(owner).createProject("my-app");
    const { high, low } = await encryptAesKey(cofheClient);
    await registry
      .connect(owner)
      .updateEnvironment(0, "production", high, low, BLOB_CID, 0);
    await registry.connect(owner).grantAccess(0, "production", member.address);
    await registry.connect(owner).revokeAccess(0, "production", member.address);

    expect(await registry.hasAccess(0, "production", member.address)).to.equal(
      false,
    );
  });

  it("rejects access grant by non-owner", async function () {
    await registry.connect(owner).createProject("my-app");
    const { high, low } = await encryptAesKey(cofheClient);
    await registry
      .connect(owner)
      .updateEnvironment(0, "production", high, low, BLOB_CID, 0);

    await expect(
      registry.connect(stranger).grantAccess(0, "production", member.address),
    ).to.be.revertedWith("Not a project owner");
  });

  it("rejects batch with >100 members", async function () {
    await registry.connect(owner).createProject("my-app");
    const { high, low } = await encryptAesKey(cofheClient);
    await registry
      .connect(owner)
      .updateEnvironment(0, "production", high, low, BLOB_CID, 0);

    const tooMany = Array(101).fill(member.address);
    await expect(
      registry.connect(owner).batchGrantAccess(0, "production", tooMany),
    ).to.be.revertedWith("Max 100 members per batch");
  });
});
```

```typescript
const ALGO = "AES-GCM";
const KEY_LENGTH = 256;
const IV_LENGTH = 12; // 96-bit IV for AES-GCM

/// Generate a fresh 256-bit AES key. Returns raw bytes as Uint8Array.
export async function generateAesKey(): Promise<Uint8Array> {
  const key = await crypto.subtle.generateKey(
    { name: ALGO, length: KEY_LENGTH },
    true, // extractable
    ["encrypt", "decrypt"],
  );
  const raw = await crypto.subtle.exportKey("raw", key);
  return new Uint8Array(raw);
}

/// Encrypt plaintext string with AES-256-GCM.
/// Returns base64(iv + ciphertext) suitable for IPFS storage.
export async function aesEncrypt(
  plaintext: string,
  keyBytes: Uint8Array,
): Promise<string> {
  const iv = crypto.getRandomValues(new Uint8Array(IV_LENGTH));
  const cryptoKey = await crypto.subtle.importKey(
    "raw",
    keyBytes,
    ALGO,
    false,
    ["encrypt"],
  );
  const encoded = new TextEncoder().encode(plaintext);
  const ciphertext = await crypto.subtle.encrypt(
    { name: ALGO, iv },
    cryptoKey,
    encoded,
  );
  const combined = new Uint8Array(IV_LENGTH + ciphertext.byteLength);
  combined.set(iv, 0);
  combined.set(new Uint8Array(ciphertext), IV_LENGTH);
  return btoa(String.fromCharCode(...combined));
}

/// Decrypt base64(iv + ciphertext) blob with AES-256-GCM.
/// Returns plaintext string.
export async function aesDecrypt(
  base64Blob: string,
  keyBytes: Uint8Array,
): Promise<string> {
  const combined = Uint8Array.from(atob(base64Blob), (c) => c.charCodeAt(0));
  const iv = combined.slice(0, IV_LENGTH);
  const ciphertext = combined.slice(IV_LENGTH);
  const cryptoKey = await crypto.subtle.importKey(
    "raw",
    keyBytes,
    ALGO,
    false,
    ["decrypt"],
  );
  const decrypted = await crypto.subtle.decrypt(
    { name: ALGO, iv },
    cryptoKey,
    ciphertext,
  );
  return new TextDecoder().decode(decrypted);
}

/// Split 32-byte AES key into two 16-byte uint128 values (high, low).
/// Returns [bigint, bigint] for FHE encryption.
export function splitAesKeyToUint128(keyBytes: Uint8Array): [bigint, bigint] {
  if (keyBytes.length !== 32) throw new Error("AES key must be 32 bytes");
  const high = keyBytes.slice(0, 16);
  const low = keyBytes.slice(16, 32);
  const toUint128 = (b: Uint8Array) =>
    b.reduce((acc, byte) => (acc << 8n) | BigInt(byte), 0n);
  return [toUint128(high), toUint128(low)];
}

/// Reconstruct 32-byte AES key from two uint128 bigint values.
export function joinUint128ToAesKey(high: bigint, low: bigint): Uint8Array {
  const toBytes = (n: bigint): Uint8Array => {
    const bytes = new Uint8Array(16);
    for (let i = 15; i >= 0; i--) {
      bytes[i] = Number(n & 0xffn);
      n >>= 8n;
    }
    return bytes;
  };
  const key = new Uint8Array(32);
  key.set(toBytes(high), 0);
  key.set(toBytes(low), 16);
  return key;
}
```

```typescript
const PINATA_GATEWAY = "https://gateway.pinata.cloud/ipfs";

/// Upload a string blob to IPFS via Pinata. Returns the CID.
export async function uploadToIPFS(
  content: string,
  name: string,
): Promise<string> {
  const jwt = process.env.NEXT_PUBLIC_PINATA_JWT;
  if (!jwt) throw new Error("NEXT_PUBLIC_PINATA_JWT not set");

  const blob = new Blob([content], { type: "application/octet-stream" });
  const formData = new FormData();
  formData.append("file", blob, name);
  formData.append(
    "pinataMetadata",
    JSON.stringify({ name: `fheenv-${name}-${Date.now()}` }),
  );

  const res = await fetch("https://api.pinata.cloud/pinning/pinFileToIPFS", {
    method: "POST",
    headers: { Authorization: `Bearer ${jwt}` },
    body: formData,
  });

  if (!res.ok) throw new Error(`Pinata upload failed: ${res.statusText}`);
  const data = await res.json();
  return data.IpfsHash as string;
}

/// Fetch a blob from IPFS by CID. Returns the raw string content.
export async function fetchFromIPFS(cid: string): Promise<string> {
  const res = await fetch(`${PINATA_GATEWAY}/${cid}`);
  if (!res.ok)
    throw new Error(`IPFS fetch failed for CID ${cid}: ${res.statusText}`);
  return res.text();
}
```

```typescript
/// Parse a .env string into a key-value record.
/// Handles: comments (#), blank lines, quoted values, inline comments.
export function parseEnv(raw: string): Record<string, string> {
  const result: Record<string, string> = {};
  for (const line of raw.split("\n")) {
    const trimmed = line.trim();
    if (!trimmed || trimmed.startsWith("#")) continue;
    const eqIdx = trimmed.indexOf("=");
    if (eqIdx === -1) continue;
    const key = trimmed.slice(0, eqIdx).trim();
    let value = trimmed.slice(eqIdx + 1).trim();
    // Strip surrounding quotes
    if (
      (value.startsWith('"') && value.endsWith('"')) ||
      (value.startsWith("'") && value.endsWith("'"))
    ) {
      value = value.slice(1, -1);
    }
    // Strip inline comments (only for unquoted values)
    const commentIdx = value.indexOf(" #");
    if (commentIdx !== -1) value = value.slice(0, commentIdx).trim();
    if (key) result[key] = value;
  }
  return result;
}

/// Serialize a key-value record back to .env string format.
export function serializeEnv(record: Record<string, string>): string {
  return Object.entries(record)
    .map(([k, v]) => {
      // Quote values containing spaces or special characters
      const needsQuotes = /[\s#"'\\]/.test(v);
      return `${k}=${needsQuotes ? `"${v.replace(/"/g, '\\"')}"` : v}`;
    })
    .join("\n");
}
```

```tsx
import { useState } from "react";
import { useAccount, usePublicClient, useWalletClient } from "wagmi";

type Props = { onClose: () => void; onCreated: (id: bigint) => void };

export function CreateProjectModal({ onClose, onCreated }: Props) {

  const publicClient = usePublicClient();

  const [name, setName] = useState("");

    if (!walletClient || !publicClient || !address) return;
    if (!name.trim()) {
      setError("Enter a project name");

      setLoading(false);
```

```tsx
"use client";
import { useState } from "react";
import {
  useAccount,
  usePublicClient,
  useWalletClient,
  useReadContract,
} from "wagmi";
import { Ethers6Adapter } from "@cofhe/sdk/adapters";
import { cofheClient, Encryptable } from "@/lib/cofhe";
import { generateAesKey, aesEncrypt, splitAesKeyToUint128 } from "@/lib/aes";
import { uploadToIPFS } from "@/lib/ipfs";
import { REGISTRY_ABI, REGISTRY_ADDRESS } from "@/lib/contracts";

type Props = { projectId: bigint; envName: string };

export function PushEnvForm({ projectId, envName }: Props) {
  const { address } = useAccount();
  const publicClient = usePublicClient();
  const { data: walletClient } = useWalletClient();
  const [rawEnv, setRawEnv] = useState("");
  const [log, setLog] = useState<string[]>([]);
  const [loading, setLoading] = useState(false);

  const { data: envData } = useReadContract({
    address: REGISTRY_ADDRESS,
    abi: REGISTRY_ABI,
    functionName: "getEnvironment",
    args: [projectId, envName],
  });

  const currentVersion = envData ? (envData[3] as bigint) : 0n;

  function addLog(msg: string) {
    setLog((p) => [...p, msg]);
  }

  async function handlePush() {
    if (!walletClient || !publicClient || !address || !rawEnv.trim()) return;
    setLoading(true);
    setLog([]);

    try {
      addLog("🔑 Generating fresh AES-256 key locally...");
      const aesKey = await generateAesKey();

      addLog(
        "🔒 Encrypting .env contents with AES-256-GCM (never leaves your browser)...",
      );
      const encryptedBlob = await aesEncrypt(rawEnv, aesKey);

      addLog("📤 Uploading encrypted blob to IPFS...");
      const cid = await uploadToIPFS(encryptedBlob, `${projectId}-${envName}`);
      addLog(`✅ Uploaded to IPFS: ${cid.slice(0, 16)}...`);

      addLog("⚙️  Connecting to CoFHE...");
      const adapter = new Ethers6Adapter(walletClient as any);
      await cofheClient.connect(publicClient as any, adapter);

      addLog("🔐 Encrypting AES key with FHE (split into 2x euint128)...");
      const [keyHigh, keyLow] = splitAesKeyToUint128(aesKey);
      // .onStep() tracks ZK proof generation which can take several seconds.
      // Steps: InitTfhe → FetchKeys → Pack → Prove → Verify
      const [encHigh, encLow] = await cofheClient
        .encryptInputs([
          Encryptable.uint128(keyHigh),
          Encryptable.uint128(keyLow),
        ])
        .onStep((step, ctx) => {
          if (ctx?.isStart) addLog(`   ⏳ ZK step: ${step}...`);
          if (ctx?.isEnd) addLog(`   ✓ ${step} done (${ctx.duration}ms)`);
        })
        .execute();

      addLog("📝 Submitting to blockchain...");
      const { request } = await publicClient.simulateContract({
        address: REGISTRY_ADDRESS,
        abi: REGISTRY_ABI,
        functionName: "updateEnvironment",
        args: [projectId, envName, encHigh, encLow, cid, currentVersion],
        account: address,
      });
      const hash = await walletClient.writeContract(request);
      await publicClient.waitForTransactionReceipt({ hash });

      addLog(`✅ Secrets pushed! Tx: ${hash.slice(0, 10)}...`);
      addLog(
        "🔒 AES key stored as FHE ciphertext. Nobody can read your secrets.",
      );
      setRawEnv("");
    } catch (e: any) {
      addLog(`❌ Error: ${e.message}`);
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className="bg-gray-900 border border-gray-800 rounded-xl p-6">
      <h3 className="font-semibold mb-1">
        Push Secrets — <span className="text-indigo-400">{envName}</span>
      </h3>
      <p className="text-xs text-gray-500 mb-4">
        Paste your .env contents. Encryption happens in your browser. Nothing
        plaintext leaves your device.
      </p>
      <textarea
        value={rawEnv}
        onChange={(e) => setRawEnv(e.target.value)}
        placeholder={
          "DATABASE_URL=postgres://...\nOPENAI_KEY=sk-proj-...\nSTRIPE_SECRET=sk_live_..."
        }
        rows={8}
        className="w-full bg-gray-950 border border-gray-700 rounded-lg px-3 py-2 text-sm font-mono focus:outline-none focus:border-indigo-500 mb-4 resize-none"
      />
      <button
        onClick={handlePush}
        disabled={loading || !rawEnv.trim()}
        className="w-full bg-indigo-600 hover:bg-indigo-700 disabled:bg-gray-700 text-white py-2 rounded-lg text-sm font-medium"
      >
        {loading ? "Encrypting & Pushing..." : "🔒 Encrypt & Push"}
      </button>
      {log.length > 0 && (
        <div className="mt-4 bg-gray-950 rounded-lg p-3 text-xs font-mono text-gray-300 space-y-1 max-h-40 overflow-y-auto">
          {log.map((l, i) => (
            <p key={i}>{l}</p>
```

```tsx
import { useAccount, usePublicClient, useWalletClient } from "wagmi";

import { cofheClient, FheTypes } from "@/lib/cofhe";
import { aesDecrypt, joinUint128ToAesKey } from "@/lib/aes";
import { fetchFromIPFS } from "@/lib/ipfs";
import { parseEnv } from "@/lib/envParser";
import { REGISTRY_ABI, REGISTRY_ADDRESS } from "@/lib/contracts";
import { useReadContract } from "wagmi";

type Props = { projectId: bigint; envName: string };

export function SecretsTable({ projectId, envName }: Props) {
  const { address } = useAccount();
  const publicClient = usePublicClient();
  const { data: walletClient } = useWalletClient();
  const [secrets, setSecrets] = useState<Record<string, string> | null>(null);
  const [revealed, setRevealed] = useState<Record<string, boolean>>({});
  const [loading, setLoading] = useState(false);
  const [loadingMsg, setLoadingMsg] = useState("Decrypting...");
  const [error, setError] = useState<string | null>(null);

  const { data: envData } = useReadContract({
    address: REGISTRY_ADDRESS,
    abi: REGISTRY_ABI,
    functionName: "getEnvironment",
    args: [projectId, envName],
  });

  async function handleDecrypt() {
    if (!walletClient || !publicClient || !envData) return;
    setLoading(true);
    setError(null);
    try {
      const adapter = new Ethers6Adapter(walletClient as any);
      await cofheClient.connect(publicClient as any, adapter);
      await cofheClient.permits.getOrCreateSelfPermit();

      const [aesKeyHighHandle, aesKeyLowHandle, blobCid] = envData as [
        bigint,
        bigint,
        string,
        bigint,
      ];

      // decryptForView polls the Threshold Network — show live progress
      // so the UI doesn't appear frozen during the polling phase.
      const highPlain = await cofheClient
        .decryptForView(aesKeyHighHandle, FheTypes.Uint128)
        .onPoll(({ attemptIndex }) =>
          setLoadingMsg(
            `Waiting for Threshold Network… (attempt ${attemptIndex + 1})`,
          ),
        )
        .execute();
      setLoadingMsg("Decrypting...");
      const lowPlain = await cofheClient
        .decryptForView(aesKeyLowHandle, FheTypes.Uint128)
        .onPoll(({ attemptIndex }) =>
          setLoadingMsg(
            `Threshold Network (low key, attempt ${attemptIndex + 1})…`,
          ),
        )
        .execute();

      const aesKey = joinUint128ToAesKey(
        highPlain as bigint,
        lowPlain as bigint,
      );

      setLoadingMsg("Fetching encrypted blob from IPFS...");
      const encryptedBlob = await fetchFromIPFS(blobCid);
      setLoadingMsg("Decrypting locally...");
      const plaintext = await aesDecrypt(encryptedBlob, aesKey);
      setSecrets(parseEnv(plaintext));
    } catch (e: any) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  }

  if (!secrets) {
    return (
      <div className="bg-gray-900 border border-gray-800 rounded-xl p-6">
        <h3 className="font-semibold mb-3">
          Secrets — <span className="text-indigo-400">{envName}</span>
        </h3>
        <p className="text-xs text-gray-500 mb-4">
          Decryption happens locally in your browser using your wallet. Nothing
          is sent to any server.
        </p>
        {error && <p className="text-red-400 text-xs mb-3">{error}</p>}

          onClick={handleDecrypt}
          disabled={loading}
          className="bg-indigo-600 hover:bg-indigo-700 disabled:bg-gray-700 text-white px-4 py-2 rounded-lg text-sm font-medium"
        >
          {loading ? loadingMsg : "🔓 Decrypt Secrets"}
        </button>
      </div>

        <h3 className="font-semibold">
          Secrets — <span className="text-indigo-400">{envName}</span>

            <th className="text-left pb-2 font-medium">KEY</th>
            <th className="text-left pb-2 font-medium">VALUE</th>
          </tr>
        </thead>
        <tbody>
          {Object.entries(secrets).map(([key, value]) => (
            <tr key={key} className="border-b border-gray-800/50">
              <td className="py-2 pr-4 font-mono text-indigo-300 text-xs">

              </td>
```

```typescript
import { ethers } from "ethers";
import fs from "fs";
import path from "path";
import os from "os";

const KEYFILE_PATH = path.join(os.homedir(), ".fheenv", "wallet.json");

/// Load wallet from:
/// 1. FHEENV_PRIVATE_KEY env var (for CI/CD)
/// 2. ~/.fheenv/wallet.json keyfile (for interactive use)
export function loadWallet(rpcUrl: string): ethers.Wallet {
  const pk = process.env.FHEENV_PRIVATE_KEY;
  const provider = new ethers.JsonRpcProvider(rpcUrl);

  if (pk) {
    return new ethers.Wallet(pk, provider);
  }

  if (fs.existsSync(KEYFILE_PATH)) {
    const keyfile = JSON.parse(fs.readFileSync(KEYFILE_PATH, "utf-8"));
    return new ethers.Wallet(keyfile.privateKey, provider);
  }

  throw new Error(
    "No wallet found. Run `fheenv login` first or set FHEENV_PRIVATE_KEY.",
  );
}

/// Save a wallet's private key to the keyfile.
export function saveWallet(privateKey: string) {
  const dir = path.dirname(KEYFILE_PATH);
  if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
  fs.writeFileSync(KEYFILE_PATH, JSON.stringify({ privateKey }), {
    mode: 0o600,
  });
}

export function walletExists(): boolean {
  return fs.existsSync(KEYFILE_PATH) || !!process.env.FHEENV_PRIVATE_KEY;
}
```

```typescript
export function initCommand(program: Command) {
  program
    .command("init")
    .description("Create a new fheENV project and save config to .fheenv.json")
    .requiredOption("-n, --name <name>", "Project name")
    .option("--rpc <url>", "Sepolia RPC URL", DEFAULT_RPC)
    .option(

      process.env.PINATA_JWT || "",

      try {
        const wallet = loadWallet(opts.rpc);

        const publicClient = createPublicClient({
          chain: sepolia,
          transport: http(opts.rpc),

        const receipt = await publicClient.waitForTransactionReceipt({ hash });
```

```typescript
import { createCipheriv, createDecipheriv, randomBytes } from "crypto";

const IV_LENGTH = 12;

export function generateAesKeyNode(): Buffer {
  return randomBytes(32); // 256-bit key
}

/// Encrypt plaintext string. Returns base64(iv + ciphertext + authTag).
export function aesEncryptNode(plaintext: string, key: Buffer): string {
  const iv = randomBytes(IV_LENGTH);
  const cipher = createCipheriv("aes-256-gcm", key, iv);
  const encrypted = Buffer.concat([
    cipher.update(plaintext, "utf8"),
    cipher.final(),
  ]);
  const authTag = cipher.getAuthTag();
  const combined = Buffer.concat([iv, encrypted, authTag]);
  return combined.toString("base64");
}

/// Decrypt base64(iv + ciphertext + authTag). Returns plaintext string.
export function aesDecryptNode(base64Blob: string, key: Buffer): string {
  const combined = Buffer.from(base64Blob, "base64");
  const iv = combined.slice(0, IV_LENGTH);
  const authTag = combined.slice(combined.length - 16);
  const ciphertext = combined.slice(IV_LENGTH, combined.length - 16);
  const decipher = createDecipheriv("aes-256-gcm", key, iv);
  decipher.setAuthTag(authTag);
  return Buffer.concat([
    decipher.update(ciphertext),
    decipher.final(),
  ]).toString("utf8");
}

export function splitAesKeyToUint128Node(key: Buffer): [bigint, bigint] {
  const toUint128 = (b: Buffer) =>
    b.reduce((acc, byte) => (acc << 8n) | BigInt(byte), 0n);
  return [toUint128(key.slice(0, 16)), toUint128(key.slice(16, 32))];
}

export function joinUint128ToAesKeyNode(high: bigint, low: bigint): Buffer {
  const toBytes = (n: bigint): Buffer => {
    const b = Buffer.alloc(16);
    for (let i = 15; i >= 0; i--) {
      b[i] = Number(n & 0xffn);
      n >>= 8n;
    }
    return b;
  };
  return Buffer.concat([toBytes(high), toBytes(low)]);
}
```

```typescript
import FormData from "form-data";
import axios from "axios";

const PINATA_GATEWAY = "https://gateway.pinata.cloud/ipfs";

export async function uploadToIPFSNode(
  content: string,
  name: string,
  jwt: string,
): Promise<string> {
  const formData = new FormData();
  formData.append("file", Buffer.from(content), { filename: name });
  formData.append("pinataMetadata", JSON.stringify({ name: `fheenv-${name}` }));

  const res = await axios.post(
    "https://api.pinata.cloud/pinning/pinFileToIPFS",
    formData,
    {
      headers: { ...formData.getHeaders(), Authorization: `Bearer ${jwt}` },
    },
  );
  return res.data.IpfsHash as string;
}

export async function fetchFromIPFSNode(cid: string): Promise<string> {
  const res = await axios.get(`${PINATA_GATEWAY}/${cid}`, {
    responseType: "text",
  });
  return res.data as string;
}
```

```typescript
/// Parse .env string into a key-value record (Node.js version).
export function parseEnvNode(raw: string): Record<string, string> {
  const result: Record<string, string> = {};
  for (const line of raw.split("\n")) {
    const trimmed = line.trim();
    if (!trimmed || trimmed.startsWith("#")) continue;
    const eqIdx = trimmed.indexOf("=");
    if (eqIdx === -1) continue;
    const key = trimmed.slice(0, eqIdx).trim();
    let value = trimmed.slice(eqIdx + 1).trim();
    if (
      (value.startsWith('"') && value.endsWith('"')) ||
      (value.startsWith("'") && value.endsWith("'"))
    ) {
      value = value.slice(1, -1);
    }
    if (key) result[key] = value;
  }
  return result;
}
```
